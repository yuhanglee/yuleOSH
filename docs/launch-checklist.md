### 域名检查命令

```bash
# 检查 DNS 解析
dig yuleosh.com
dig app.yuleosh.com

# 检查 HTTPS
curl -I https://app.yuleosh.com

# 检查重定向
curl -I http://yuleosh.com

# 检查证书到期
echo | openssl s_client -connect app.yuleosh.com:443 -servername app.yuleosh.com 2>/dev/null | openssl x509 -dates -noout
```

### 备份验证脚本（示例）

```bash
#!/bin/bash
# 每周恢复验证
RESTORE_DB="yuleosh_restore_test_$(date +%Y%m%d)"
pg_restore --dbname=$RESTORE_DB /backups/latest/yuleosh_full.dump
if [ $? -eq 0 ]; then
  echo "[OK] 备份 ${RESTORE_DB} 可恢复"
  psql -c "DROP DATABASE $RESTORE_DB;"
else
  echo "[FAIL] 备份恢复失败，请立即检查!"
  curl -X POST -H "Content-Type: application/json" \
    -d '{"text":"❌ 数据库备份恢复验证失败"}' \
    [webhook_url]
fi
```
